3. Trigger the file download
Replace the window.location.href = mailto:... line with:
jsconst blob = new Blob([emlContent], { type: 'message/rfc822' });
const url = URL.createObjectURL(blob);
const a = document.createElement('a');
a.href = url;
a.download = `RFQ_${projectName}_${vendorName}.eml`.replace(/\s+/g, '_');
document.body.appendChild(a);
a.click();
document.body.removeChild(a);
URL.revokeObjectURL(url);
