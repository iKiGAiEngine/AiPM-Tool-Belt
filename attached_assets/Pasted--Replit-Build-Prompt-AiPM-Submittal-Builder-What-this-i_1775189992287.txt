# Replit Build Prompt: AiPM Submittal Builder

## What this is

You are creating a React web application called AiPM Submittal Builder. This is an internal SaaS tool for a commercial construction specialty contractor called National Building Specialties (NBS). It is part of a larger product suite called AiPM (styled with lowercase i).

The purpose of the tool is to let a project manager (PM) create submittal packages from construction estimate workbooks. The PM selects a project, uploads an estimate Excel file, and the system parses it into scope tabs and schedule line items. The PM then assigns product data PDFs to each line item, builds a cover page, validates the package, previews it, and exports it.

```javascript
// Persistent storage adapter using localStorage.
// All submittal projects are stored under keys prefixed with "submittal:".
// In a future phase this can be swapped for a real database client
// without changing any calling code.

const PREFIX = "submittal:";

function getItem(key) {
  try {
    const raw = localStorage.getItem(key);
    return raw ? JSON.parse(raw) : null;
  } catch {
    return null;
  }
}

function setItem(key, value) {
  try {
    localStorage.setItem(key, JSON.stringify(value));
    return true;
  } catch {
    return false;
  }
}

function removeItem(key) {
  try {
    localStorage.removeItem(key);
    return true;
  } catch {
    return false;
  }
}

function listKeys(prefix) {
  const results = [];
  for (let i = 0; i < localStorage.length; i++) {
    const key = localStorage.key(i);
    if (key && key.startsWith(prefix)) {
      results.push(key);
    }
  }
  return results;
}

// Load all saved submittal projects, sorted by most recently updated first.
export async function loadAllProjects() {
  const keys = listKeys(PREFIX);
  const projects = [];
  for (const k of keys) {
    const p = getItem(k);
    if (p) projects.push(p);
  }
  return projects.sort((a, b) => (b.updatedAt || 0) - (a.updatedAt || 0));
}

// Load a single project by ID.
export async function loadProject(id) {
  return getItem(PREFIX + id);
}

// Save a project. Sets updatedAt automatically.
export async function saveProject(project) {
  const p = { ...project, updatedAt: Date.now() };
  setItem(PREFIX + p.id, p);
  return p;
}

// Delete a project by ID.
export async function deleteProject(id) {
  removeItem(PREFIX + id);
}
```

```javascript
// Package validation engine.
// Audits a submittal project and returns categorized issues.

export function validateProject(project) {
  if (!project || !project.scopes) {
    return { errors: [], warnings: [], info: [], summary: {} };
  }

  const errors = [];
  const warnings = [];
  const info = [];

  let totalLines = 0;
  let attached = 0;
  let missing = 0;
  let blankCallout = 0;
  let blankDesc = 0;
  let blankModel = 0;
  let zeroQty = 0;
  let totalAttPages = 0;

  project.scopes.forEach((scope) => {
    // Check for missing cover rows
    if (!scope.coverLines || scope.coverLines.length === 0) {
      warnings.push({ scope: scope.tabName, msg: "No cover page rows defined" });
    }

    const scopeCallouts = {};
    const scopeModels = {};

    scope.lines.forEach((line) => {
      totalLines++;

      // Skip lines marked not required or by others
      if (line.lineStatus === "not_required" || line.lineStatus === "by_others") {
        info.push({
          scope: scope.tabName,
          line: line.callout || "(no callout)",
          msg: "Marked " + line.lineStatus.replace("_", " "),
        });
        return;
      }

      // Attachment check
      const attCount = line.attachments ? line.attachments.length : 0;
      const attPages = line.attachments
        ? line.attachments.reduce((a, x) => a + (x.pageCount || 0), 0)
        : 0;
      totalAttPages += attPages;

      if (attCount > 0) {
        attached++;
      } else {
        missing++;
        warnings.push({
          scope: scope.tabName,
          line: line.callout || "(blank)",
          msg: "Missing product data",
        });
      }

      // Blank field checks
      if (!line.callout || !line.callout.trim()) {
        blankCallout++;
        errors.push({
          scope: scope.tabName,
          line: "Row " + line.sortOrder,
          msg: "Blank callout",
        });
      }

      if (!line.desc || !line.desc.trim()) {
        blankDesc++;
        errors.push({
          scope: scope.tabName,
          line: line.callout || "Row " + line.sortOrder,
          msg: "Blank description",
        });
      }

      if (!line.model || !line.model.trim()) {
        blankModel++;
        warnings.push({
          scope: scope.tabName,
          line: line.callout || "(blank)",
          msg: "Blank model number",
        });
      }

      const qtyVal = Number(line.qty);
      if (!line.qty || qtyVal === 0 || isNaN(qtyVal)) {
        zeroQty++;
        warnings.push({
          scope: scope.tabName,
          line: line.callout || "(blank)",
          msg: "Zero or missing quantity",
        });
      }

      // Track duplicates within scope
      if (line.callout && line.callout.trim()) {
        const key = line.callout.trim();
        scopeCallouts[key] = (scopeCallouts[key] || 0) + 1;
      }
      if (line.model && line.model.trim()) {
        const key = line.model.trim();
        scopeModels[key] = (scopeModels[key] || 0) + 1;
      }
    });

    // Report duplicates
    Object.entries(scopeCallouts)
      .filter(([, count]) => count > 1)
      .forEach(([key, count]) => {
        info.push({
          scope: scope.tabName,
          msg: "Duplicate callout: " + key + " (" + count + "x)",
        });
      });

    Object.entries(scopeModels)
      .filter(([, count]) => count > 1)
      .forEach(([key, count]) => {
        info.push({
          scope: scope.tabName,
          msg: "Duplicate model: " + key + " (" + count + "x)",
        });
      });
  });

  // Projected total pages: cover + schedule pages + attachment pages per scope
  const schedulePages = project.scopes.reduce((a, s) => {
    return a + Math.max(1, Math.ceil((s.lines ? s.lines.length : 0) / 15));
  }, 0);
  const projectedPages = project.scopes.length + schedulePages + totalAttPages;

  return {
    errors,
    warnings,
    info,
    summary: {
      totalScopes: project.scopes.length,
      totalLines,
      attached,
      missing,
      blankCallout,
      blankDesc,
      blankModel,
      zeroQty,
      totalAttPages,
      projectedPages,
    },
  };
}
```

```javascript
// Pagination engine for submittal packages.
// Computes page numbers for cover, schedule (multi-page), and attachments.

export const LINES_PER_SCHEDULE_PAGE = 15;

export function computePagination(scope) {
  if (!scope) {
    return {
      cover: 1,
      scheduleStart: 2,
      scheduleEnd: 2,
      schedulePages: 1,
      attachments: [],
      total: 2,
    };
  }

  const lineCount = scope.lines ? scope.lines.length : 0;
  const schedulePages = Math.max(1, Math.ceil(lineCount / LINES_PER_SCHEDULE_PAGE));
  const scheduleEnd = 1 + schedulePages;

  let page = scheduleEnd + 1;
  const attachments = [];

  if (scope.lines) {
    scope.lines.forEach((line) => {
      if (line.attachments) {
        line.attachments.forEach((att) => {
          const pgCount = att.pageCount || 1;
          attachments.push({
            id: att.id,
            fileName: att.fileName,
            callout: line.callout,
            model: line.model,
            calloutStamp: att.calloutStamp,
            pageCount: pgCount,
            startPage: page,
            endPage: page + pgCount - 1,
          });
          page += pgCount;
        });
      }
    });
  }

  return {
    cover: 1,
    scheduleStart: 2,
    scheduleEnd: scheduleEnd,
    schedulePages: schedulePages,
    attachments: attachments,
    total: attachments.length > 0 ? page - 1 : scheduleEnd,
  };
}
```

```javascript
  var [activeScopeIdx, setActiveScopeIdx] = useState(0);

  useEffect(function () {
    loadProject(projectId).then(function (p) {
      if (p) {
        setProject(p);
        setActiveTab(p.lastActiveTab || "schedule");
        if (p.lastActiveScopeId && p.scopes) {
          var idx = p.scopes.findIndex(function (s) {
            return s.id === p.lastActiveScopeId;
          });
          if (idx >= 0) setActiveScopeIdx(idx);
        }
      }
      setLoading(false);
    });
  }, [projectId]);

  // Autosave with debounce
  var triggerSave = useCallback(
    function (updated) {
      setSaving("unsaved");
      clearTimeout(saveTimer.current);
      saveTimer.current = setTimeout(function () {
        setSaving("saving");
        var scope =
          updated.scopes && updated.scopes[activeScopeIdx]
            ? updated.scopes[activeScopeIdx]
            : null;
        var toSave = Object.assign({}, updated, {
          lastActiveScopeId: scope ? scope.id : null,
          lastActiveTab: activeTab,
          lastOpenedAt: now(),
        });

        // Compute completion
        var totalLines = 0;
        var attachedLines = 0;
        if (toSave.scopes) {
          toSave.scopes.forEach(function (s) {
            if (s.lines) {
              totalLines += s.lines.length;
              s.lines.forEach(function (l) {
                if (l.attachments && l.attachments.length > 0) attachedLines++;
              });
            }
          });
        }
        toSave.completionPercent = totalLines > 0 ? pct(attachedLines, totalLines) : 0;

        // Compute status
        if (!toSave.scopes || toSave.scopes.length === 0) {
          toSave.submittalStatus = "not_started";
        } else if (attachedLines === 0) {
          toSave.submittalStatus = "in_progress";
        } else if (attachedLines < totalLines) {
          toSave.submittalStatus = "waiting_product_data";
        } else {
          toSave.submittalStatus = "ready_for_review";
        }

        saveProject(toSave).then(function () {
          setSaving("saved");
        });
      }, 800);
    },
```

```javascript
  const editLine = (lineId, field, value) => {

      if (line) line[field] = value;
      return p;
    });

        attachments: [],

      return p;

      );
      return p;
    });

        </span>
```

```javascript
                  <th style={thS}>Type</th>
                  <th style={thS}>Comments</th>
                </tr>
              </thead>
              <tbody>
                {scope.coverLines &&

                        <td style={tdS}>{c.type}</td>

                      <td style={tdS} />
                      <td style={tdS} />
                      <td style={tdS} />
                    </tr>
                  );
                })}
              </tbody>
            </table>
          </div>
        </PageFrame>

        {/* Schedule pages */}
        {schedulePages.map(function (sp, si) {
          return (
            <PageFrame
              key={si}
              num={sp.pageNum}
              total={pageInfo.total}
              label={si === 0 ? "SCHEDULE" : "SCHEDULE (cont.)"}
            >
              <div style={{ padding: "16px 20px", textAlign: "center" }}>

                    width: "100%",

                  }}
                >
                  <thead>
                    <tr style={{ background: "#f0f0f0" }}>
                      <th style={thS}>SPEC No.</th>
                      <th style={thS}>SPEC TITLE</th>
                      <th style={thS}>CALLOUT</th>

                      <th style={thS}>MODEL</th>
                      <th style={thS}>QTY</th>

                  <tbody>

                            {l.model}
                          </td>
                          <td style={tdS}>{l.qty}</td>
                        </tr>
                      );
                    })}
                  </tbody>
                </table>
              </div>
            </PageFrame>
          );
        })}

        {/* Product data pages */}
        {pageInfo.attachments.map(function (a) {
          return (
            <PageFrame
              key={a.id}
              num={a.startPage}
              total={pageInfo.total}
              label={"PRODUCT DATA \u2014 " + a.callout}
            >
              <div style={{ padding: 20, textAlign: "center", position: "relative" }}>
                <div
                  style={{
                    position: "absolute",
                    top: 6,
                    right: 10,
                    padding: "2px 8px",
                    borderRadius: 3,
                    background: "#BF9B30",
                    color: "#000",
                    fontSize: 8,
                    fontWeight: 800,
                  }}
                >
                  {a.callout}
                </div>
                <div style={{ fontSize: 32, color: "#ccc", margin: "30px 0 10px" }}>
                  &#128196;
                </div>
                <div style={{ fontSize: 11, fontWeight: 600, color: "#333" }}>
                  {a.fileName}
                </div>
                <div style={{ fontSize: 9, color: "#888", marginTop: 3 }}>
                  {a.pageCount} page{a.pageCount > 1 ? "s" : ""} &middot; {a.model}
                </div>
              </div>
            </PageFrame>
          );
        })}
      </div>
    </div>
  );
}
```
