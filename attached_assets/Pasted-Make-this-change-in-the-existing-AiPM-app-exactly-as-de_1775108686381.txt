function projectMatchesDashboardScope(p, currentUser) {
  const scope = currentUser?.dashboardScope || '';

  if (scope === 'company_wide') {
    return true;
  }

  if (scope === 'my_projects') {
    const myInitials = currentUser?.initials || '';
    return !!myInitials && p.nbsEstimator === myInitials;
  }

  if (scope === 'my_region') {
    const assignedRegion = currentUser?.assignedRegion || '';
    return !!assignedRegion && p.region === assignedRegion;
  }

  const role = currentUser?.role || '';
  const myInitials = currentUser?.initials || '';

  if (role === 'admin' && myInitials === 'HK') {
    return !!myInitials && p.nbsEstimator === myInitials;
  }

  if (myInitials) {
    return p.nbsEstimator === myInitials;
  }

  return false;
}
